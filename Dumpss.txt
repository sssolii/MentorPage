    <script>
        // Get the image element by its id
        const rotatingImage = document.getElementById('rotatingImage');
    
        // Store the initial rotation
        const initialRotation = -8;
    
        // Add an event listener for the 'scroll' event
        window.addEventListener('scroll', function () {
            // Adjust the rotation speed by multiplying the scroll position
            const rotationSpeed = 0.1; // Adjust this value to control the rotation speed
            const rotationAngle = window.scrollY * rotationSpeed % 360;
    
            // Calculate the final rotation, ensuring it doesn't go below the initial rotation
            const finalRotation = Math.max(rotationAngle - initialRotation, initialRotation);
    
            // Apply the rotation to the image around the y-axis
            rotatingImage.style.transform = `translate(-50%, -50%) rotate(${finalRotation}deg)`;
        });

        let next = document.querySelector('.next')
        let prev = document.querySelector('.prev')

        next.addEventListener('click', function(){
            let items = document.querySelectorAll('.item')
             document.querySelector('.slide').appendChild(items[0])
        })

        prev.addEventListener('click', function(){
             let items = document.querySelectorAll('.item')
             document.querySelector('.slide').prepend(items[items.length - 1]) // here the length of items = 6
        })
    </script>
